import { Component, Input, OnInit } from '@angular/core';
import {
  fadeInOnEnterAnimation,
  fadeOutOnLeaveAnimation,
  bounceOutOnLeaveAnimation,
} from 'angular-animations';

export interface NavbarItem {
  type: 'INTERNAL' | 'EXTERNAL';
  path: string;
  icon: string;
}

@Component({
  selector: 'authdare-navbar-item',
  templateUrl: './navbar-item.component.html',
  styleUrls: ['./navbar-item.component.css'],
  animations: [
    fadeInOnEnterAnimation(),
    fadeOutOnLeaveAnimation(),
    bounceOutOnLeaveAnimation(),
  ],
})
export class NavbarItemComponent implements OnInit {
  @Input() navbarItem!: NavbarItem;
  @Input() isVisible = true;
  @Input() index = 1000;
  constructor() {}

  ngOnInit(): void {
    this.index++;
  }
}

<ng-container *ngIf="navbarItem.type == 'INTERNAL'">
  <button
    mat-fab
    [style.animationDuration]="index * 500 + 'ms'"
    [routerLink]="navbarItem.path"
    routerLinkActive="router-link-active"
    *ngIf="isVisible"
    [@fadeOutOnLeave]
    [@bounceOutOnLeave]
  >
    <mat-icon *ngIf="navbarItem.icon"> {{ navbarItem.icon }} </mat-icon>
  </button>
</ng-container>
<ng-container *ngIf="navbarItem.type == 'EXTERNAL'">
  <a
    mat-fab
    [style.animationDuration]="index * 500 + 'ms'"
    [href]="navbarItem.path"
    *ngIf="isVisible"
    [@fadeOutOnLeave]
    [@bounceOutOnLeave]
  >
    <mat-icon *ngIf="navbarItem.icon"> {{ navbarItem.icon }} </mat-icon>
  </a>
</ng-container>
